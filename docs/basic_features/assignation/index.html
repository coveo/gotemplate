Assignation within expression

@-foreach ($value := to(10))
    @{value}
@-end foreach

@-foreach ($index, $value := to(10))
    @{index} = @($value * 2)
@-end foreach

@-if ($result := 2+2 == 4)
    result = @{result}
@-end if

@-with ($value := 2+2)
    value = @{value}
@-end with
